AVALIAÇÔES 

CREATE TABLE avaliacoes (
    id INT NOT NULL AUTO_INCREMENT,
    nota INT NOT NULL,
    data_avaliacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (id)
);

CLIENTE

CREATE TABLE cliente (
    id INT NOT NULL AUTO_INCREMENT,
    email VARCHAR(255) NOT NULL UNIQUE,
    nome VARCHAR(255) NOT NULL,
    telefone VARCHAR(255),
    compras TEXT,
    PRIMARY KEY (id)
);

CONTRATOS

CREATE TABLE contratos (
    id INT NOT NULL AUTO_INCREMENT,
    descricao VARCHAR(255),
    status VARCHAR(20),
    data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (id)
);

DEPARTAMENTO CREATE TABLE departamento (
    id VARCHAR(255) NOT NULL,
    nome VARCHAR(255) NOT NULL,
    PRIMARY KEY (id)
);

FUNCIONARIOS

CREATE TABLE funcionarios (
    id INT NOT NULL AUTO_INCREMENT,
    nome VARCHAR(255),
    cargo VARCHAR(255),
    salario DOUBLE,
    email VARCHAR(255),
    senha VARCHAR(255),
    loja VARCHAR(50),
    lojaId VARCHAR(255),
    PRIMARY KEY (id),
    KEY loja_fk (lojaId)
);

LOJA

CREATE TABLE loja (
    id VARCHAR(50) NOT NULL,
    nome VARCHAR(100),
    planoAtual VARCHAR(50),
    PRIMARY KEY (id)
);

NOTA FISCAL

CREATE TABLE nota_fiscal (
    id INT NOT NULL AUTO_INCREMENT,
    chave_acesso VARCHAR(36) NOT NULL,
    email_cliente VARCHAR(255) NOT NULL,
    titulo_id VARCHAR(36) NOT NULL,
    valor_pago DOUBLE,
    data_emissao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (id),
    FOREIGN KEY (titulo_id) REFERENCES titulo(id) -- Presumindo que haja uma tabela chamada 'titulo' com uma coluna 'id'
);

LOGS

CREATE TABLE operation_logs (
    id INT NOT NULL AUTO_INCREMENT,
    funcionario_id VARCHAR(255),
    nome VARCHAR(255),
    cargo VARCHAR(255),
    operacao VARCHAR(255),
    timestamp DATETIME DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (id)
);

PRODUTO

CREATE TABLE produto (
    id VARCHAR(255) NOT NULL,
    nome VARCHAR(255) NOT NULL,
    preco DOUBLE NOT NULL,
    departamento VARCHAR(255),
    PRIMARY KEY (id),
    FOREIGN KEY (departamento) REFERENCES departamento(id)
);


RELATÓRIO VENDAS

CREATE TABLE relatorio_vendas (
    id INT NOT NULL AUTO_INCREMENT,
    departamento VARCHAR(255) UNIQUE,
    nome_departamento VARCHAR(255),
    total_acumulado DOUBLE DEFAULT 0,
    PRIMARY KEY (id)
);

SUGESTÕES

CREATE TABLE sugestoes (
    id INT NOT NULL AUTO_INCREMENT,
    sugestao TEXT NOT NULL,
    data_sugestao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (id)
);

TITULO

CREATE TABLE titulo (
    id VARCHAR(36) NOT NULL,
    quantidade DOUBLE,
    paga TINYINT(1),
    departamento VARCHAR(50),
    email_cliente VARCHAR(255),
    lojaId INT,
    PRIMARY KEY (id),
    FOREIGN KEY (email_cliente) REFERENCES clientes(email) ON DELETE SET NULL,
    FOREIGN KEY (lojaId) REFERENCES loja(id) ON DELETE SET NULL
);
